Var
   // Seção de Declarações das variáveis
   n1: real

Inicio
   escreval ("Digite um numero")
   leia (n1)

   se (n1 mod 2) = 0 entao
      escreval (" o numero é par " )
   senao
      escreval (" o numero é ímpar ")
   fimSe
